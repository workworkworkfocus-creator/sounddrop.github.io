<script>

function loadSong() {

    let baseUrl = document.getElementById("urlInput").value.trim();

    if(!baseUrl) return alert("Paste your song URL");

    let mp3Url = baseUrl + ".mp3";

    document.getElementById("playerSection").innerHTML = `

        <audio controls src="${mp3Url}"></audio>

        <div>

            <button onclick="downloadFile('${baseUrl}.mp3')">MP3</button>

            <button onclick="downloadFile('${baseUrl}.wav')">WAV</button>

            <button onclick="downloadFile('${baseUrl}-lossless.flac')">Lossless</button>

            <button onclick="downloadFile('${baseUrl}.mp4')">MP4</button>

            <button onclick="downloadFile('${baseUrl}-instrumental.mp3')">Instrumental</button>

            <button onclick="downloadFile('${baseUrl}-vocals.mp3')">Vocals</button>

            <button onclick="downloadFile('${baseUrl}-stems.zip')">Stems</button>

        </div>

    `;

    saveHistory(baseUrl);

}

function downloadFile(url) {

    const a = document.createElement("a");

    a.href = url;

    a.download = "";

    document.body.appendChild(a);

    a.click();

    document.body.removeChild(a);

}

function saveHistory(url) {

    let history = JSON.parse(localStorage.getItem("downloads") || "[]");

    if(!history.includes(url)) {

        history.push(url);

        localStorage.setItem("downloads", JSON.stringify(history));

    }

    showHistory();

}

function showHistory() {

    let history = JSON.parse(localStorage.getItem("downloads") || "[]");

    let list = document.getElementById("historyList");

    list.innerHTML = "";

    history.forEach(item => {

        let li = document.createElement("li");

        li.textContent = item;

        list.appendChild(li);

    });

}

showHistory();

</script>
